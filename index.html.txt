<script>
function calculateResult(){

const scores = {
hero:0,
tech:0,
leader:0,
helper:0
};

for(let i=1;i<=10;i++){
const answer=document.querySelector(`input[name="q${i}"]:checked`);

if(!answer){
alert("Kérlek válaszolj minden kérdésre!");
return;
}

scores[answer.value]++;
}

let winner = Object.keys(scores).reduce((a,b)=>
scores[a] > scores[b] ? a : b
);

const result = document.getElementById("result");

if(winner==="hero"){
result.style.background="#ffe5e5";
result.innerHTML=`
<h2>🚒 Beavatkozó Hős</h2>
<p>Bátor vagy, gyorsan cselekszel és szívesen segítesz másoknak.</p>
<div class="prize">🎁 Ajándékod: Kulcstartó + Oklevél</div>
<h3>Mutasd meg ezt az eredményt a tűzoltóknak!</h3>
`;
}

if(winner==="tech"){
result.style.background="#e5f4ff";
result.innerHTML=`
<h2>🔧 Műszaki Mentő</h2>
